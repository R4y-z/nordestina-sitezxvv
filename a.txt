version: "3.8"

services:
  postgres:
    image: postgres:16-alpine
    container_name: nordestina_postgres
    restart: unless-stopped
    environment:
      POSTGRES_USER: nordestina_user
      POSTGRES_PASSWORD: nordestina_pass_dev
      POSTGRES_DB: nordestina_restaurante
    ports:
      - "5432:5432"
    volumes:
      - pgdata:/var/lib/postgresql/data
    networks:
      - sz_network

  redis:
    image: redis:7-alpine
    container_name: sz_redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    networks:
      - sz_network

  api:
    build:
      context: ./apps/api
      dockerfile: Dockerfile.dev
    container_name: sz_api
    restart: unless-stopped
    environment:
      DATABASE_URL: postgresql://nordestina_user:nordestina_pass_dev@postgres:5432/nordestina_restaurante
      REDIS_URL: redis://redis:6379
      JWT_SECRET: change_in_production_super_secret_key
      JWT_EXPIRES_IN: 7d
      PORT: 3001
      NODE_ENV: development
    ports:
      - "3001:3001"
    volumes:
      - ./apps/api:/app
      - /app/node_modules
    depends_on:
      - postgres
      - redis
    networks:
      - sz_network

volumes:
  pgdata:

networks:
  sz_network:
    driver: bridge
